	valid_relation_trigger = {
		FROM = {

				ROOT = { is_ai = yes }
				FROM = { is_ai = yes }
			}
			NOT = { has_global_flag = sheep_selected }
		}
	}

	license_purchase_cost = 0 #base cost reduction
	ai_license_acceptance = 200 #more to AI acceptance value.
}
