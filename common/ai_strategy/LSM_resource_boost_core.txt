
ENG_infra_resource_boost_steel_130 = {
	allowed = {
		original_tag = ENG
	}
	enable = {
		OR = {
			check_variable = { resource_exported@steel > 120 }
			check_variable = { resource_imported@steel > 80 }
		}
		130 = {
			free_building_slots = {
				building = infrastructure
				size > 0
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = build_building
		id = infrastructure
		target = 130 
		value = 2000
	}
}
