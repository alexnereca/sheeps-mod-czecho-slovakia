
benelux_only_hold_europe = {
	allowed = {
		OR = {
			original_tag = BEL
			original_tag = LUX
			original_tag = HOL
		}
	}
	enable = { 
		has_capitulated = no
		OR = {
			AND = {
				original_tag = BEL
				NOT = {
					any_country = {
						original_tag = GER
						is_in_faction_with = BEL
					}
				}
			}
			AND = {
				original_tag = HOL
				NOT = {
					any_country = {
						original_tag = GER
						is_in_faction_with = HOL
					}
				}
			}
			AND = {
				original_tag = LUX
				NOT = {
					any_country = {
						OR = {
							original_tag = GER
							original_tag = ENG
							original_tag = FRA
						}
						is_in_faction_with = LUX
					}
				}
			}
		}
	}

	abort_when_not_enabled = yes
	ai_strategy = {
		type = front_unit_request
		state_trigger = { 
			NOT = { is_on_continent = europe }
		}
		value = -300
	}

	ai_strategy = {
		type = force_defend_ally_borders
		value = 1000
		target = europe
	}

	ai_strategy = {
		type = dont_defend_ally_borders
		value = 1000
		target = africa
	}

	ai_strategy = {
		type = dont_defend_ally_borders
		value = 1000
		target = middle_east
	}

	ai_strategy = {
		type = dont_defend_ally_borders
		value = 1000
		target = south_america
	}

	ai_strategy = {
		type = dont_defend_ally_borders
		value = 1000
		target = north_america
	}

	ai_strategy = {
		type = dont_defend_ally_borders
		value = 1000
		target = asia
	}
}


BEL_defend_HOL = {
	allowed = {
		original_tag = BEL
	}
	enable = {
		is_historical_focus_on = yes
		HOL = { has_government = democratic }
		BEL = { has_government = democratic }
		GER = { 
			OR = {
				has_completed_focus = GER_around_maginot
				focus_progress = {
					focus = GER_around_maginot
					progress > 0.00	
				}
				is_justifying_wargoal_against = HOL
				has_wargoal_against = HOL
			}
			is_ai = no
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = protect
		id = "HOL"
		value = 500
	}
}
