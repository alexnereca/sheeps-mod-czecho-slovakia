
GER_USA_grab_basic_medium = {
	allowed = {
		OR = {
			original_tag = GER
			original_tag = USA
		}
	}
	enable = {
		NOT = { has_tech = basic_medium_tank }
	}

	abort = {
		has_tech = basic_medium_tank
	}

	ai_strategy = {
		type = research_tech
		id = basic_medium_tank
		value = 100
	}
}
