	}
	enable = {
		is_historical_focus_on = yes
		SPR = {
			has_completed_focus = SPA_la_division_azul
			has_civil_war = no
			has_army_manpower = { size > 600000 }
		}
		OR = {
			original_tag = FRA
			original_tag = ITA
			original_tag = GER
		}
		OR = {
			has_government = fascism
			has_government = neutrality
		}
		NOT = {
			any_enemy_country = {
				has_government = fascism
				is_major = yes
			}
		}
		is_in_faction_with = GER
		OR = {
			has_global_flag = allied_victory_in_africa
			date > 1942.1.1
		}
		OR = {
			date > 1943.1.1
			SPR = { has_completed_focus = SPA_ensure_agricultural_self_sufficiency }
		} 
		any_enemy_country = {
			is_ai = no
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = send_volunteers_desire
