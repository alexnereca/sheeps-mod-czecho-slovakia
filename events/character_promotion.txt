country_event = {
	id = character_promotion.3
	title = character_promotion.3.t
	desc = character_promotion.3.d
	picture = GFX_report_event_german_politician_speech
	
	trigger = {
		original_tag = USA
		OR = {
			AND = {
				has_government = communism
				has_idea = earl_browder
			}
			AND = {
				OR = {
					has_government = neutrality	
					has_government = fascism
				}
				has_idea = charles_coughlin
			}
		}
		OR = {
			has_civil_war = no
			any_enemy_country = {
				is_major = yes
				NOT = { original_tag = USA }
			}
		}
	}

	fire_only_once = yes

	mean_time_to_happen = { days = 7 }
	
	option = {
		name = character_promotion.3.a
		if = {
			limit = { 
				has_idea = charles_coughlin
			}
			USA_charles_coughlin = {
				remove_trait = {
					trait = fascist_demagogue
					slot = political_advisor
				}
				add_trait = {
					trait = USA_fascist_revolutionary
					slot = political_advisor
				}
			}
		}
		if = {
			limit = { 
				has_idea = earl_browder
			}
			USA_earl_browder = {
				remove_trait = {
					trait = communist_revolutionary
					slot = political_advisor 
				}
				add_trait = {
					trait = USA_syndicalist_vanguard
					slot = political_advisor 
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
}
